#include<conio.h>
int main()
{
    int i,j,n,k,p,ch;
    printf("enter numbers of lines:");
    scanf("%d",&n);
    for(i=1;i<=n;i++){
        ch=64;
        for(j=1;j<=10-2*i;j++){
            printf(" ");
        }
        for(k=1;k<=i;k++){
            ch+=1;
            printf("%c ",ch);
        }
        for(p=1;p<=i-1;p++){
            ch-=1;
            printf("%c ",ch);
        }
        printf("\n");
    }
    getch();
}
